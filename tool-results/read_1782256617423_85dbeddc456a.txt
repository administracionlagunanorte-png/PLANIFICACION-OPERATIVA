   435→// Helper: adjust a date to the nearest weekday (Mon-Fri)
   436→// If Saturday -> move to Monday, if Sunday -> move to Monday
   437→function adjustToWeekday(date: Date): Date {
   438→  const d = new Date(date)
   439→  const dow = d.getDay()
   440→  if (dow === 0) { // Sunday -> Monday
   441→    d.setDate(d.getDate() + 1)
   442→  } else if (dow === 6) { // Saturday -> Monday
   443→    d.setDate(d.getDate() + 2)
   444→  }
   445→  return d
   446→}
   447→
   448→// Helper: check if a date is a weekday (Mon-Fri)
   449→function isWeekday(date: Date): boolean {
   450→  const dow = date.getDay()
   451→  return dow >= 1 && dow <= 5
   452→}
   453→
   454→// Helper: get the first business day (Mon-Fri) of a month
   455→function getFirstBusinessDayOfMonth(year: number, month: number): Date {
   456→  let d = 1
   457→  let date = new Date(year, month, d)
   458→  while (!isWeekday(date)) {
   459→    d++
   460→    date = new Date(year, month, d)
   461→  }
   462→  return date
   463→}
   464→
   465→// Helper: get the nth occurrence of a specific weekday in a month
   466→function getNthWeekdayOfMonth(year: number, month: number, weekday: number, n: number): Date {
   467→  const firstDay = new Date(year, month, 1)
   468→  let firstWeekday = firstDay.getDay()
   469→  let diff = weekday - firstWeekday
   470→  if (diff < 0) diff += 7
   471→  const firstOccurrence = 1 + diff
   472→  if (n === 4) {
   473→    // Last occurrence - find last
   474→    const lastDay = new Date(year, month + 1, 0).getDate()
   475→    let lastOccurrence = firstOccurrence
   476→    while (lastOccurrence + 7 <= lastDay) {
   477→      lastOccurrence += 7
   478→    }
   479→    return new Date(year, month, lastOccurrence)
   480→  }
   481→  return new Date(year, month, firstOccurrence + (n - 1) * 7)
   482→}
   483→
   484→// Helper: get all Mondays (or any weekday) in a month
   485→function getWeekdaysInMonth(year: number, month: number, weekday: number): Date[] {
   486→  const dates: Date[] = []
   487→  const firstDay = new Date(year, month, 1)
   488→  let firstWeekday = firstDay.getDay()
   489→  let diff = weekday - firstWeekday
   490→  if (diff < 0) diff += 7
   491→  const firstOccurrence = 1 + diff
   492→  const lastDay = new Date(year, month + 1, 0).getDate()
   493→  for (let d = firstOccurrence; d <= lastDay; d += 7) {
   494→    dates.push(new Date(year, month, d))
   495→  }
   496→  return dates
   497→}

   499→// POST /api/mantenimiento-generate — Auto-generate LVs for a month
   500→export async function POST(req: NextRequest) {
   501→  try {
   502→    const body = await req.json()
   503→    const { year, month } = body // month is 0-indexed (0=January)
   504→
   505→    if (year === undefined || month === undefined) {
   506→      return NextResponse.json({ error: 'year y month son obligatorios' }, { status: 400 })
   507→    }
   508→
   509→    const y = Number(year)
   510→    const m = Number(month)
   511→    const daysInMonth = new Date(y, m + 1, 0).getDate()
   512→    const monthStart = new Date(y, m, 1)
   513→    const monthEnd = new Date(y, m, daysInMonth, 23, 59, 59)
   514→
   515→    let created = 0
   516→    let skipped = 0
   517→
   518→    for (const template of LV_TEMPLATES) {
   519→      // Determine all scheduled dates for this template in this month
   520→      const scheduledDates: Date[] = []
   521→
   522→      switch (template.scheduleRule) {
   523→        case 'daily': {
   524→          // Every weekday (Mon-Fri) of the month — NO weekends
   525→          for (let d = 1; d <= daysInMonth; d++) {
   526→            const date = new Date(y, m, d)
   527→            if (isWeekday(date)) {
   528→              scheduledDates.push(date)
   529→            }
   530→          }
   531→          break
   532→        }
   533→        case '3x_week': {
   534→          // Monday, Wednesday, Friday
   535→          for (let d = 1; d <= daysInMonth; d++) {
   536→            const date = new Date(y, m, d)
   537→            const dow = date.getDay()
   538→            if (dow === 1 || dow === 3 || dow === 5) {
   539→              scheduledDates.push(date)
   540→            }
   541→          }
   542→          break
   543→        }
   544→        case 'weekly': {
   545→          const wd = template.weekDay ?? 1 // Default Monday
   546→          scheduledDates.push(...getWeekdaysInMonth(y, m, wd))
   547→          break
   548→        }
   549→        case 'biweekly': {
   550→          // Days 1 and 15 of the month, adjusted to nearest weekday if weekend
   551→          const d1 = adjustToWeekday(new Date(y, m, 1))
   552→          scheduledDates.push(d1)
   553→          if (daysInMonth >= 15) {
   554→            const d15 = adjustToWeekday(new Date(y, m, 15))
   555→            scheduledDates.push(d15)
   556→          }
   557→          break
   558→        }
   559→        case 'monthly': {
   560→          // First business day of the month
   561→          const targetDate = getFirstBusinessDayOfMonth(y, m)
   562→          scheduledDates.push(targetDate)
   563→          break
   564→        }
   565→        case 'quarterly': {
   566→          const qMonths = template.quarterMonths ?? []
   567→          if (qMonths.includes(m + 1)) {
   568→            const targetDate = getFirstBusinessDayOfMonth(y, m)
   569→            scheduledDates.push(targetDate)
   570→          }
   571→          break
   572→        }
   573→        case 'semiannual': {
   574→          const sMonths = template.semiMonths ?? []
   575→          if (sMonths.includes(m + 1)) {
   576→            const targetDate = getFirstBusinessDayOfMonth(y, m)
   577→            scheduledDates.push(targetDate)
   578→          }
   579→          break
   580→        }
   581→        case 'annual': {
   582→          const aMonth = template.annualMonth ?? 1
   583→          if (m + 1 === aMonth) {
   584→            const targetDate = getFirstBusinessDayOfMonth(y, m)
   585→            scheduledDates.push(targetDate)
   586→          }
   587→          break
   588→        }
   589→      }
   590→
   591→      // For each scheduled date, check if LV already exists, if not create it
   592→      for (const date of scheduledDates) {
   593→        // Use local date components to avoid timezone issues
   594→        const localYear = date.getFullYear()
   595→        const localMonth = String(date.getMonth() + 1).padStart(2, '0')
   596→        const localDay = String(date.getDate()).padStart(2, '0')
   597→        const dateStr = `${localYear}-${localMonth}-${localDay}`
   598→
   599→        // Check if an LV with this codigo and date already exists
   600→        const existing = await db.mantenimientoLV.findFirst({
   601→          where: {
   602→            codigo: template.codigo,
   603→            scheduledDate: {
   604→              gte: new Date(dateStr + 'T00:00:00.000Z'),
   605→              lte: new Date(dateStr + 'T23:59:59.999Z'),
   606→            },
   607→          },
   608→        })
   609→
   610→        if (existing) {
   611→          skipped++
   612→          continue
   613→        }
   614→
   615→        // Create the LV with the date set to noon local to avoid timezone shift
   616→        const scheduledDate = new Date(dateStr + 'T12:00:00.000Z')
   617→
   618→        await db.mantenimientoLV.create({
   619→          data: {
   620→            codigo: template.codigo,
   621→            nombre: template.nombre,
   622→            sector: template.sector,
   623→            frecuencia: template.frecuencia,
   624→            scheduledDate,
   625→            status: 'PENDIENTE',
   626→            progress: 0,
   627→            items: {
   628→              create: template.items.map((item, idx) => ({
   629→                category: item.category,
   630→                description: item.description,
   631→                status: 'PENDIENTE',
   632→                order: idx,
   633→              })),
   634→            },
   635→          },
   636→        })
   637→        created++
   638→      }
   639→    }
   640→
   641→    return NextResponse.json({
   642→      success: true,
   643→      month: m + 1,
   644→      year: y,
   645→      created,
   646→      skipped,
   647→      message: `Se crearon ${created} listas de verificación (${skipped} ya existían)`,
   648→    })
   649→  } catch (error) {
   650→    console.error('Error generating mantenimiento LVs:', error)
   651→    return NextResponse.json({ error: 'Error al generar listas de verificación' }, { status: 500 })
   652→  }
   653→}
